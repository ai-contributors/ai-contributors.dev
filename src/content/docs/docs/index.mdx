import docsConfig from '@/docs-config';
import { SPEC } from '../../../data/spec.generated';
import { url as siteUrl } from '../../../lib/links';

export const DOCS = docsConfig.docs;
export const byKey = new Map(DOCS.map((entry) => [entry.key, entry]));
export const hrefForKey = (key) => {
  const entry = byKey.get(key);
  if (!entry) return '#';
  return siteUrl(entry.path);
};
export const docTitle = (entry) => entry.title ?? entry.label;

export const PILLARS = SPEC.pillars.length;
export const CLAUSES = SPEC.clauses.length;
export const LEVELS = SPEC.levels.length;
export const VERSION = SPEC.version;
export const LEVEL_RANGE = `${SPEC.levels[0].id}-${SPEC.levels[SPEC.levels.length - 1].id}`;

export const featured = [
  {
    key: 'quickstart',
    kind: 'Adopt',
    description:
      'Choose an audit path, install the skills, run the first audit, and review the generated artifacts.',
    foot: 'first audit',
  },
  {
    key: 'spec',
    kind: 'Read',
    description: `${CLAUSES} clauses across ${PILLARS} pillars. The normative document: every MUST, MUST when applicable, SHOULD, and MAY.`,
    foot: `spec rev ${VERSION}`,
  },
  {
    key: 'audit-prompt',
    kind: 'Audit',
    description:
      'Drop a no-install audit prompt into a tool-using agent and produce stamped artifacts.',
    foot: 'prompt path',
  },
  {
    key: 'ref-levels',
    kind: 'Reference',
    description: `${LEVELS} levels, ${LEVEL_RANGE}. Each level adds one gate on top of the previous.`,
    foot: 'level ladder',
  },
];

export const sections = DOCS.reduce((acc, entry) => {
  let bucket = acc[acc.length - 1];
  if (!bucket || bucket.label !== entry.section) {
    bucket = { label: entry.section, items: [] };
    acc.push(bucket);
  }
  bucket.items.push(entry);
  return acc;
}, []);

and descriptions come from the rule catalog projection.

## Conformance levels

The level ladder describes how much of the workflow an AI tool
participates in. Each level is cumulative.

<div class="levels-table">
  <div class="lvl-row lvl-head">
    <span>Min. level</span>
    <span>Name</span>
    <span>Autonomy</span>
    <span>Workflow</span>
    <span>Definition</span>
  </div>
  {SPEC.levels.map((level) => {
    const pct =
      SPEC.levels.length <= 1
        ? 0
        : Math.round((level.order / (SPEC.levels.length - 1)) * 100);
    return (
      <div class="lvl-row" data-level={level.id}>
        <span class="lvl-tag mono">{level.id}</span>
        <span class="lvl-name">{level.label}</span>
        <span class="lvl-auto">
          <span class="lvl-bar"><span class="lvl-fill" style={`width: ${pct}%`} /></span>
          <span class="mono small">{pct}%</span>
        </span>
        <span class="lvl-desc">{level.workflowSummary}</span>
        <span class="lvl-review">{level.description}</span>
      </div>
    );
  })}
</div>

## How to read these docs

Use the [Quickstart](./quickstart/) if you want to run an audit now. Use
the [Adoption guide](./adoption-guide/) if you are changing a repository.
Use the [Rule catalog](./rule-catalog/) if you are building tooling. Use
[Published pages](./doc-coverage/) to inspect what this site publishes
and where each page comes from.

## Versioning and stability

The site renders the pinned specification revision. Audit artifacts pin
the spec revision they were stamped against, and the validator
cross-checks that stamp. See [Releases](./releases/) for the version
timeline.

:::note[docs vs. spec]
If a guide or page in these docs disagrees with the
<a href={siteUrl('specification')}>full specification</a>, the
specification wins.
:::
