import { SPEC } from '../../../data/spec.generated';
import { url as link } from '../../../lib/links';

export const pillars = SPEC.pillars.map((pillar, index) => {
  const clauses = SPEC.clauses.filter((clause) => clause.pillar === pillar.id);
  const ruleCount = clauses.reduce((count, clause) => count + clause.rules.length, 0);
  return {
    ...pillar,
    index: index + 1,
    clauses,
    ruleCount,
  };
});

## The seven pillars

The pillar names, descriptions, icons, and clause ranges below come from
`AI-CONTRIBUTOR-RULE-CATALOG.json`. This page owns the presentation only.

## Clause coverage

Each pillar is a contiguous clause range in the catalog. Follow the clause
links to the full specification for the rule text and audit level.
